    def plot_bar(df, col, style='absolute', order=None, want_num_labels=True, fig=None, ax=None, palette='viridis', verbose=True):
    """`plot_bar`

    Plot a bar chart of the column of interest either in absolute or relative numbers.

    Parameters
    ----------
    df : pandas DataFrame
        DataFrame to be plotted.

    col : str
        Column of the DataFrame to be plotted.

    style : str
        String that defines the style. Possible values are:

        * `'absolute'` or `'n'` - Number of batches
        * `'relative'` or `'p'` - Percentage of batches

    order : list of str
        Order to plot. If `None`, order is inferred from the data.

    want_num_labels : bool
        Boolean that defines if bars will have number labels (`True`)
        or not (`False`).

    fig : matplotlib figure handle
        Figure handle.

    ax : matplotlib ax handle
        Axes handle.

    palette : str
        Any string that can be interpreted by seaborn [`color_palette`](https://seaborn.pydata.org/generated/seaborn.color_palette.html#seaborn.color_palette)
        (including any of [`matplotlib`'s color maps'](https://matplotlib.org/stable/users/explain/colors/colormaps.html)).
        Some examples are `'viridis'`, `'winter'`, `'jet'`. 

        !!! tip 
            Notice you can reverse any palette by appending `'_r'` at the end 
            of the name (e.g., `'viridis_r'`).

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    fig, ax : tuple
        For further customization outside of the function.

    Displays
    --------
    Bar chart.

    See also
    ----------
    * See [`kipy.plot_bar_by_date`][] for a bar plot with a time dimension.
    """
    if verbose:
        print(f"Generating bar chart of {col}...", end='', flush=False)

    # Create figure (if necessary).
    if (fig is None) and (ax is None):
        fig, ax = plt.subplots(1, 1, figsize=[8, 6])
    elif fig is None:
        fig = ax.get_figure()
    elif ax is None:
        ax = fig.gca()

    df_plot = df.copy()

    # Check that columns are in the given DataFrame.
    if col not in df_plot.columns:
        raise Exception(f"Column {col} not present in DataFrame")

    # Actually plot.
    if (style=='absolute') or (style=='n'):
        sns.countplot(x=df_plot[col], ax=ax, palette='viridis')

    elif (style=='relative') or (style=='p'):        
        df_p = df[col].value_counts(normalize=True).reset_index().rename({'index':col, col:'percent'}, axis=1)
        sns.barplot(x=col, y='percent', data=df_p, palette='viridis')


    else:
        raise Exception(f"{style} is not a valid plot style.")

    ax.spines['top'].set_visible(False)
    ax.spines['right'].set_visible(False)

    # Add counts to each bar.
    # See https://stackoverflow.com/a/68334380/948768
    # and https://stackoverflow.com/a/76241039/948768
    if want_num_labels:
        if (style=='absolute') or (style=='n'):
            ax.bar_label(ax.containers[0])
        elif (style=='relative') or (style=='p'):
            labels = [f'{(h)*100:0.1f}%' if (h := v.get_height()) > 0 else '' for v in ax.containers[0]]
            ax.bar_label(ax.containers[0], labels=labels)


    ax.set_xlabel(col.replace('_', ' ').capitalize())

    # Format x-axis tick labels so that x-ticks are integers.
    # See https://stackoverflow.com/a/50045907/948768
    if col in ['harvest_day', 'number_bags']:
        xlabels = [item.get_text() for item in ax.get_xticklabels()]
        ax.set_xticklabels([str(round(float(label))) for label in xlabels])

    # Format y-axis tick labels so that y-ticks correspond to 0-100%.
    if (style=='relative') or (style=='p'):
        ylabels = [item.get_text() for item in ax.get_yticklabels()]
        ax.set_yticks(ax.get_yticks().tolist()) # To avoid warning. See https://stackoverflow.com/a/68794383/948768
        ax.set_yticklabels([str(round(float(label)*100)) for label in ylabels])

    if (style=='absolute') or (style=='n'):
        ax.set_ylabel("Number of Batches")
    elif (style=='relative') or (style=='p'):
        ax.set_ylabel("Proportion of Batches [%]")

    if verbose:
        print("\tDONE!")

    return fig, ax

    def plot_bar_by_date(df, col, col_date, style='absolute', want_num_labels=True, fig=None, ax=None, palette='viridis', verbose=True):
    """`plot_bar_by_date`

    Plot a bar chart of the column of interest ordered by date either in absolute or relative numbers.

    Parameters
    ----------
    df : pandas DataFrame
        DataFrame to be plotted.

    col : str
        Column of the DataFrame to be plotted.

    col_date : str
        Column of the DataFrame with date information. 

    style : str
        String that defines the style. Possible values are:

        * `'absolute'` or `'n'` - Number of batches
        * `'relative'` or `'p'` - Percentage of batches

    want_num_labels : bool
        Boolean that defines if bars will have number labels (`True`)
        or not (`False`).

    fig : matplotlib figure handle
        Figure handle.

    ax : matplotlib ax handle
        Axes handle.

    palette : str
        Any string that can be interpreted by seaborn [`color_palette`](https://seaborn.pydata.org/generated/seaborn.color_palette.html#seaborn.color_palette)
        (including any of [`matplotlib`'s color maps'](https://matplotlib.org/stable/users/explain/colors/colormaps.html)).
        Some examples are `'viridis'`, `'winter'`, `'jet'`. 

        !!! tip 
            Notice you can reverse any palette by appending `'_r'` at the end 
            of the name (e.g., `'viridis_r'`).

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    fig, ax : tuple
        For further customization outside of the function.

    Displays
    --------
    Bar chart with time dimension.

    See also
    ----------
    * See [`kipy.plot_bar`][] for a bar plot of all batches (with no time dimension).
    """
    if verbose:
        print(f"Generating bar chart (by date) of {col}...", end='', flush=False)

    # Create figure (if necessary).
    if (fig is None) and (ax is None):
        fig, ax = plt.subplots(1, 1, figsize=[8, 6])
    elif fig is None:
        fig = ax.get_figure()
    elif ax is None:
        ax = fig.gca()


    # Check that columns are in the given DataFrame.
    for col_ in [col, col_date]:
        if col_ not in df.columns:
            raise Exception(f"Column {col_} not present in DataFrame")


    if (style=='absolute') or (style=='n'):
        df_grouped = df.groupby(col_date)[col].value_counts(normalize=False).unstack(col)
    elif (style=='relative') or (style=='p'):
        df_grouped = df.groupby(col_date)[col].value_counts(normalize=True).unstack(col)
    else:
        raise Exception(f"{style} is not a valid plot style.")

    df_grouped.plot.bar(stacked=True, cmap=cm.get_cmap(palette), width=0.8, ax=ax)


    # Format x-axis tick labels
    xlabels = ax.get_xticklabels()
    ax.set_xticklabels(xlabels, rotation=25, ha='right', rotation_mode='anchor')

    ax.spines['top'].set_visible(False)
    ax.spines['right'].set_visible(False)
    ax.legend(bbox_to_anchor=(1.04, 0.95), title=col.replace('_', ' ').capitalize(), loc="upper left", frameon=False)

    # Format legend (make sure it displays integers)
    leg = ax.get_legend()
    for lbl in leg.get_texts():
        label_text = lbl.get_text()
        new_text = f"{float(label_text):.0f}"
        lbl.set_text(new_text)


    # Format y-axis tick labels so that y-ticks correspond to 0-100%.
    if (style=='relative') or (style=='p'):
        ylabels = [item.get_text() for item in ax.get_yticklabels()]
        ax.set_yticks(ax.get_yticks().tolist()) # To avoid warning. See https://stackoverflow.com/a/68794383/948768
        ax.set_yticklabels([str(round(float(label)*100)) for label in ylabels])

        ax.set_ylim([0, 1.05])

    ax.set_xlabel("Date")

    if (style=='absolute') or (style=='n'):
        ax.set_ylabel("Number of Batches")
    elif (style=='relative') or (style=='p'):
        ax.set_ylabel("Proportion of Batches [%]")

    # Add numeric labels.
    if want_num_labels:
        for idx, row in df_grouped.reset_index(drop=True).iterrows():
            cumulative = 0

            for element in row:
                # Check to avoid NaNs
                if element == element:

                    if (style=='absolute') or (style=='n'):
                        ax.text(idx, cumulative + element / 2,
                            f"{element:.0f}",
                            va='center',
                            ha='center')
                    elif (style=='relative') or (style=='p'):
                        ax.text(idx, cumulative + element / 2,
                            f"{element * 100:.1f}%",
                            va='center',
                            ha='center')
                    cumulative += element

    if verbose:
        print("\tDONE!")

    return fig, ax

    def plot_distribution(df, col, want_metrics=False, tolerance_int=None, fig=None, ax=None, verbose=True):
    """`plot_distribution`

    Plot the distribution of a DataFrame column.

    Parameters
    ----------
    df : pandas DataFrame
        DataFrame to be plotted.

    col : str
        Column name of `df` 

    want_metrics : bool
        * If `True`, basic descriptive metrics (mean, STD, min, max, 
        percentiles) are displayed in the plot (as a legend). 
        * If `False`, none are displayed.

    tolerance_int : int or str
        Coverage of the tolerance intervals defined as follows:

        * `67` or `'67'`: 1 STD, thus covering 67% of the data (assuming normality)
        * `95` or `'95'`: 2 STD, thus covering 95% of the data (assuming normality)
        * `99` or `'99'`: 3 STD, thus covering 99% of the data (assuming normality)
        * `0` or `'0'` or `None`: Tolerance intervals are not plotted.

    fig : matplotlib figure handle
        Figure handle.

    ax : matplotlib ax handle
        Axes handle.

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    fig, ax : tuple
        For further customization outside of the function.

    Displays
    --------
    Distribution plot.
    """

    if verbose:
        print(f"Generating distribution plot of column {col}...", end='', flush=False)

    # Create figure (if necessary).
    if (fig is None) and (ax is None):
        fig, ax = plt.subplots(1, 1, figsize=[8, 6])
    elif fig is None:
        fig = ax.get_figure()
    elif ax is None:
        ax = fig.gca()


    # Check that column is in the given DataFrame.
    if col not in df.columns:
        raise Exception(f"Column {col} not present in DataFrame")


    # Create basic plot.
    sns.histplot(data=df, x=col, kde=True, ax=ax)

    # Get the original xlim (to set it again later)
    xlim = ax.get_xlim()

    # Descriptive statistics.
    # In order to show the descriptive metrics in the plot, we will
    # "plot" them as invisible lines and add them to the legend.
    if want_metrics:
        metrics = df[col].describe()

        for param, value in metrics.items():

            # We skip count, since it isn't interesting (and plotting it
            # makes no sense).
            if param == 'count':
                continue

            ax.axvline(value, color=[1, 1, 1], alpha=0, linestyle='--', label=f"{param.capitalize()} = {value:.2f}")


    # Plot tolerance interval.
    # Now, we define it as the mean +/- 2STD
    allowed_tolerances = [68, '68', 95, '95', 99, '99']
    no_tolerances = [0, '0', None]
    if tolerance_int in allowed_tolerances:
        metrics = df[col].describe()
        mean = metrics.loc['mean']
        std = metrics.loc['std']

        # Indicate the number of STD to include in the tolerance intervals
        if (tolerance_int == 68) or (tolerance_int == '68'):
            n_std = 1
            tolerance_str = '68'
        if (tolerance_int == 95) or (tolerance_int == '95'):
            n_std = 2
            tolerance_str = '95'
        if (tolerance_int == 99) or (tolerance_int == '99'):
            n_std = 3
            tolerance_str = '99'

        tolerance_int = [mean - (n_std*std), mean + (n_std*std)]
        ax.axvspan(xmin=tolerance_int[0], xmax=tolerance_int[1], color='g', alpha=0.1, label=f"TI ({tolerance_str}%) = {tolerance_int[0]:.2f} - {tolerance_int[1]:.2f}")


    elif tolerance_int in no_tolerances:
        pass
    else:
        print("\nInvalid tolerance interval. Tolerance intervals will not be plotted.")


    # Show the legend.
    if want_metrics or tolerance_int:
        ax.legend(bbox_to_anchor=(1.04, 0.95), title=col.replace('_', ' ').capitalize(), loc='upper left', frameon=False)

    # Set xlim back to the original
    # Since we "plot" the lines of the metrics (just so that they appear
    # on the legend), we need to switch back the x-lim to the original.
    ax.set_xlim(xlim)

    ax.spines['top'].set_visible(False)
    ax.spines['right'].set_visible(False)

    ax.set_xlabel(col.replace('_', ' ').capitalize())
    ax.set_ylabel("No. batches")

    # plt.show()

    if verbose:
        print("\tDONE!")

    return fig, ax

    sns.boxplot(data=df, x=col, ax=ax_box)
    ax_box.set_xlabel('')
    ax_box.spines['top'].set_visible(False)
    ax_box.spines['right'].set_visible(False)
    ax_box.spines['left'].set_visible(False)

    plot_distribution(df, col, want_metrics=want_metrics, tolerance_int=tolerance_int, fig=None, ax=ax_dist, verbose=False)

    fig.tight_layout(h_pad=3)

    # plt.show()

    if verbose:
        print("\tDONE!")

    return fig, [ax_box, ax_dist]

    Visualization of missing values.
