    def add_engineered_columns(df, product, verbose=True):
    """`add_engineered_columns`

    Add a set of engineered columns to a given `CLP`/`XLP` DataFrame.

    !!! info
        All products get the same new columns. The only difference is
        *how* these are calculated. More specifically, the `production_date`
        definition is different between YESCARTA and TECARTUS. For
        more information, see [`kipy.add_production_date`][]

    Parameters
    ----------
    df : pandas DataFrame
        Original DataFrame.

    product : str
        The product of interest. It accepts the following values:

        * `yescarta` or `clp`
        * `tecartus` or `xlp`

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    df : pandas DataFrame
        Same as the input DataFrame but with added columns

        * `production_date`
        * `production_date_numeric`
        * `production_date_formatted`
        * `production_date_q_formatted`
        * `production_date_q`
        * `production_date_quarter`
        * `production_date_month`
        * `production_date_month_name`
        * `production_date_year`
        * `production_date_short`
        * `production_date_short_formatted`
        * `harvested`
        * `harvest_day`
        * `attempt_number`
        * `attempt_type`
        * `verified`
        * `success`
        * `tcell_min_expansion_viability`

    """

    # Since verbose is handled internally for each function, there's 
    # no need to add any extra prints here.
    df_production_date = add_production_date(df, product, verbose=verbose)
    df_production_dates = add_production_date_columns(df_production_date, verbose=verbose)
    df_harvested = add_harvested(df_production_dates, verbose=verbose)
    df_harvest_day = add_harvest_day(df_harvested, verbose=verbose)
    df_attempts = add_attempt_number_type(df_harvest_day, verbose=verbose)
    df_verified = add_verified(df_attempts, verbose=verbose)
    df_tcell = add_tcell_min_expansion_viability(df_verified, verbose=verbose)
    df_engineered = add_success(df_tcell, verbose=verbose)

    return df_engineered

    add_production_date

    Add production_date column.

    Info

    In the past, production_date was defined as harvest_date. If the batch had no harvest_date (e.g., if the batch was terminated before harvest), we used start_day0_date.

    However, this definition is no longer valid. Now, the definition of production_date is different for YESCARTA and TECARTUS, as shown in the following table.

    In the future, the YESCARTA definition might align with that from TECARTUS, but it will depend on the results of the discussion regarding that column (still ongoing).

    Danger

    If the definition of production_date changes in the future, this function needs to be updated.

    def add_production_date(df, product, verbose=True):
    """`add_production_date`

    Add `production_date` column.

    !!! info
        *In the past*, `production_date` was defined as `harvest_date`.
        If the batch had no `harvest_date` (e.g., if the batch was terminated
        before harvest), we used `start_day0_date`. 

        However, this definition is no longer valid. Now, the definition of 
        `production_date` is different for YESCARTA and TECARTUS, as shown 
        in the following table.

        | Product  | `production_date` definition |
        | ---------| -----------------------------|
        | YESCARTA | Given by the so-called "JMP scripts logic": find the estimated processing duration (in days) and add it to `start_day0_date`. |
        | TECARTUS | Given by the column `harvest_termination_date` |

        In the future, the YESCARTA definition *might* align with that from
        TECARTUS, but it will depend on the results of the discussion 
        regarding that column (still ongoing).

    !!! danger
        If the definition of `production_date` changes in the future, this 
        function needs to be updated.

    Parameters
    ----------
    df : pandas DataFrame
        Original DataFrame    

    product : str
        The product of interest. It accepts the following values:

        * `yescarta` or `clp`
        * `tecartus` or `xlp`

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    df_production_date : pandas DataFrame
        Original DataFrame with the column `production_date` added.

    See also
    --------
    * [`kipy.add_production_date_columns`][]
    """
    df_production_date = df.copy(deep=True)

    if 'production_date' in df_production_date.columns:
        if verbose:
            print(". Column `production_date` already present in df.")

    else:
        if (product == 'clp') or (product == 'yescarta'):
            if verbose:
                print("+ Adding column `production_date`...", end='', flush=True)

            # Estimate the process duration based on finding the last day from
            # which we have batch data.
            def get_process_duration(row):
                if (~np.isnan(row['volume_adjusted_day10'])) or (~np.isnan(row['viability_day10'])):
                    process_duration = 10
                elif (~np.isnan(row['volume_adjusted_day9'])) or (~np.isnan(row['viability_day9'])):
                    process_duration = 9
                elif (~np.isnan(row['volume_adjusted_day8'])) or (~np.isnan(row['viability_day8'])):
                    process_duration = 8
                elif (~np.isnan(row['volume_adjusted_day7'])) or (~np.isnan(row['viability_day7'])):
                    process_duration = 7
                elif (~np.isnan(row['culture_volume_day6'])) or (~np.isnan(row['viability_day6'])):
                    process_duration = 6
                elif (~np.isnan(row['culture_volume_day5'])) or (~np.isnan(row['viability_day5'])):
                    process_duration = 5
                elif (~np.isnan(row['bag_volume_total_expansion'])) or (~np.isnan(row['expansion_viability_day3'])):
                    process_duration = 3
                elif (~np.isnan(row['transduction_postseeding_density'])) or (~np.isnan(row['viability_wash1'])):
                    process_duration = 2
                else:
                    process_duration = 0

                return process_duration

            df['process_duration'] = df.apply(get_process_duration, axis=1)
            df['process_duration'] = pd.to_timedelta(df['process_duration'], unit='D')

            # Calculate production date as `start_day0_date` + `process_duration`.
            df['production_date'] = pd.to_datetime(df['start_day0_date'], dayfirst=True) + df['process_duration']

            df_production_date['production_date'] = df['production_date'].dt.strftime('%d-%b-%y')

            if verbose:
                print("\tDONE!")


        elif (product == 'xlp') or (product == 'tecartus'):
            if not 'harvest_termination_date' in df.columns:
                raise Exception("Column harvest_termination_date not present in DataFrame.")
            else:
                if verbose:
                    print("+ Adding column `production_date`...", end='', flush=True)
                df_production_date.loc[:, 'production_date'] = df['harvest_termination_date']
                if verbose:
                    print("\tDONE!")

    return df_production_date

    def add_production_date_columns(df, col_production_date='production_date', verbose=True):
    """`add_production_date_columns`

    Add formatted production date columns to a DataFrame.

    !!! example

        If production_date is `15-Feb-22`, the following columns will be added:

        | Column name                       | Description                                                        | Output                |
        | --------------------------------- | -------------------------------------------------------------------|-----------------------|
        | `production_date_numeric`         | Production date in numeric format                                  | `15-02-22`            |
        | `production_date_formatted`       | Production date in proper datetime format                          | `2022-02-15 00:00:00` |
        | `production_date_q_formatted`     | Production date as quarter in proper period format                 | `2022Q1`              |
        | `production_date_q`               | Production date as quarter                                         | `2022-q1`             |
        | `production_date_quarter`         | Production date quarter number                                     | `1`                   |
        | `production_date_month`           | Production month as a string with leading zero if necessary        | `02`                  |
        | `production_date_month_name`      | Production month name                                              | `'February'`          |
        | `production_date_year`            | Production year as a string                                        | `2022`                |
        | `production_date_short`           | Production date with only year and month as string                 | `2022-02`             |
        | `production_date_short_formatted` | Production date with only year and month in proper datetime format | `2022-02-01 00:00:00` |

    Parameters
    ----------
    df : pandas DataFrame
        Original DataFrame

    col_production_date : str
        DataFrame column where the production date information is stored.
        This column should be in the format `DD-MMM-YY` (as stored
        in the `CLP`/`XLP` files). 

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    df_production_date : pandas DataFrame
        DataFrame with the extra columns. See Examples.

    See also
    --------
    * [`kipy.add_production_date`][]
    """

    # Check that col_production_date is present in the DataFrame.
    # If not, raise an exception.
    if col_production_date not in df.columns:
        raise Exception(f"Column {col_production_date} is not present in df.")


    df_production_date = df.copy()

    if 'production_date_numeric' not in df.columns:
        if verbose:
            print("+ Adding column `production_date_numeric`...", end='', flush=True)
        df_production_date['production_date_numeric'] = df_production_date[col_production_date].apply(lambda x: pd.to_datetime(x).strftime('%d-%m-%y') if not pd.isnull(x) else np.nan)
        if verbose:
            print("\tDONE!")
    else:
        if verbose:
            print(". Column `production_date_numeric` already present in the DataFrame")

    if 'production_date_formatted' not in df.columns:
        if verbose:
            print("+ Adding column `production_date_formatted`...", end='', flush=True)
        df_production_date['production_date_formatted'] = df_production_date['production_date_numeric'].apply(lambda x: pd.to_datetime(x, dayfirst=True) if not pd.isnull(x) else np.nan)
        if verbose:
            print("\tDONE!")
    else:
        if verbose:
            print(". Column `production_date_formatted` already present in the DataFrame")            

    if 'production_date_q_formatted' not in df.columns:
        if verbose:
            print("+ Adding column `production_date_q_formatted`...", end='', flush=True)
        df_production_date['production_date_q_formatted'] = df_production_date['production_date_formatted'].apply(lambda x: x.to_period('Q') if not pd.isnull(x) else np.nan)
        if verbose:
            print("\tDONE!")
    else:
        if verbose:
            print(". Column `production_date_q_formatted` already present in the DataFrame")    

    if 'production_date_q' not in df.columns:
        if verbose:
           print("+ Adding column `production_date_q`...", end='', flush=True)
        df_production_date['production_date_q'] = df_production_date['production_date_formatted'].apply(lambda x: str(x.year) + '-q' + str(x.quarter) if not pd.isnull(x) else np.nan)
        if verbose:
            print("\tDONE!")
    else:
        if verbose:
            print(". Column `production_date_q` already present in the DataFrame")  

    if 'production_date_quarter' not in df.columns:
        if verbose:
            print("+ Adding column `production_date_quarter`...", end='', flush=True)
        df_production_date['production_date_quarter'] = df_production_date['production_date_q_formatted'].apply(lambda x: str(x.quarter) if not pd.isnull(x) else np.nan)
        if verbose:
            print("\tDONE!")
    else:
        if verbose:
            print(". Column `production_date_quarter` already present in the DataFrame")  

    if 'production_date_month' not in df.columns:
        if verbose:
            print("+ Adding column `production_date_month`...", end='', flush=True)
        df_production_date['production_date_month'] = df_production_date['production_date_formatted'].apply(lambda x: str(x.month).zfill(2) if not pd.isnull(x) else np.nan) # Ensure month has leading 0.
        if verbose:
            print("\tDONE!")
    else:
        if verbose:
            print(". Column `production_date_month` already present in the DataFrame")  

    if 'production_date_month_name' not in df.columns:
        if verbose:
            print("+ Adding column `production_date_month_name`...", end='', flush=True)
        def _month_number_to_name(row):
            if pd.isna(row['production_date_month']):
                month_name = '-'
            else:
                month_name = calendar.month_name[int(row['production_date_month'])]
            return month_name
        df_production_date['production_date_month_name'] = df_production_date.apply(_month_number_to_name, axis=1)    
        if verbose:
            print("\tDONE!")
    else:
        if verbose:
            print(". Column `production_date_month_name` already present in the DataFrame") 

    if 'production_date_year' not in df.columns:
        if verbose:
            print("+ Adding column `production_date_year`...", end='', flush=True)
        df_production_date['production_date_year'] = df_production_date['production_date_formatted'].apply(lambda x: str(x.year) if not pd.isnull(x) else np.nan)
        if verbose:
            print("\tDONE!")
    else:
        if verbose:
            print(". Column `production_date_year` already present in the DataFrame") 

    if 'production_date_short' not in df.columns:
        if verbose:
            print("+ Adding column `production_date_short`...", end='', flush=True)
        df_production_date['production_date_short'] = df_production_date['production_date_formatted'].apply(lambda x: str(x.year) + '-' + str(x.month).zfill(2) if not pd.isnull(x) else np.nan)
        if verbose:
            print("\tDONE!")
    else:
        if verbose:
            print(". Column `production_date_short` already present in the DataFrame") 

    if 'production_date_short_formatted' not in df.columns:
        if verbose:
            print("+ Adding column `production_date_short_formatted`...", end='', flush=True)
        df_production_date['production_date_short_formatted'] = df_production_date['production_date_short'].apply(lambda x: pd.to_datetime(x, format='%Y-%m') if not pd.isnull(x) else np.nan)
        if verbose:
            print("\tDONE!")
    else:
        if verbose:
            print(". Column `production_date_short_formatted` already present in the DataFrame") 

    return df_production_date

    def add_success(df, col_status='status', verbose=True):
    """`add_success`

    Classify if a batch was successful (`True`) or not (`False`) based on the 
    status information of the given column.

    !!! danger

        If the definition of `success` changes in the future, this function 
        needs to be updated.

    Parameters
    ----------
    df : pandas DataFrame
        Original DataFrame

    col_status : str
        DataFrame column where the status date is stored.

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    df: pandas DataFrame
        Same as the input DataFrame but with an added column `success`.
    """
    if 'success' in df.columns:
        if verbose:
            print(". Column `success` already present in df.")

    else:
        if col_status not in df.columns:
            raise Exception(f"Column {col_status} is not present in df.")

        if verbose:
            print("+ Adding column `success`...", end='', flush=True)

        def _classify_success(status):

            if pd.isnull(status):
                success = np.nan
            else:
                curr_status = status.lower().strip()
                if (curr_status == 'release') or (curr_status == 'physician\'s release'):
                    success = True
                else:
                    success = False

            return success

        df['success'] = df.apply(lambda x: _classify_success(x[col_status]), axis=1)

        if verbose:
            print("\tDONE!")

    return df

    def add_harvest_day(df, verbose=True):

        Notice that this only applies to YESCARTA (`CLP`), since TECARTUS 
        (`XLP`) already has the `harvest_day` column.

    Parameters
    ----------
    df : pandas DataFrame
        Original DataFrame

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    df : pandas DataFrame.
        Same as the input DataFrame but with an added column `harvest_day`.
    """

    # Check if harvest_day already exists (which is the case in XLP)
    if 'harvest_day' in df.columns:
        if verbose:
            print(". Column `harvest_day` already present in df.")
    else:        
        if verbose:
            print("+ Adding column `harvest_day`...", end='', flush=True)

        # These are the necessary columns to compute harvest_day.
        volume_columns = ['volume_adjusted_day7', 'volume_adjusted_day8', 'volume_adjusted_day9', 'volume_adjusted_day10']

        if not set(volume_columns).issubset(df.columns):
            # If the necessary columns are not present, raise exception.
            raise Exception("Volume columns (volume_adjusted_dayX) are not present in the df.")
        else:
            # If the necessary columns are present, actually compute harvest_day.
            df['harvest_day'] = (df[volume_columns].count(axis=1)) + 6

            # If a batch had a harvest_day of 6, it means that all the
            # volume_adjusted_dayX columns were empty. This means that
            # the batch was actually not harvested. Thus, we need
            # to fill those values with NaNs.
            df.loc[df['harvest_day'] == 6, 'harvest_day'] = np.nan

        if verbose:
            print("\tDONE!")

    return df

    def add_attempt_number_type(df, verbose=True):
    """`add_attempt_number_type`

    Add the attempt number and attempt type to a DataFrame.

    !!! info
        There are two reasons why a batch could have gone through more than one
        attempt. These (and their corresponding original definition) are: 

        * Re-manufacture (`reman`): batches that share the same `subject_id`, 
            but have slightly different `batch_id`, changing only the first 
            character (e.g., `a483965890` and `d483965890`).

        * Re-apheresis (`reaph`): batches that share the same `subject_id`, 
            but have completely different `batch_id` (e.g., `a483965890` and `a529326491`)

        However, identifying them using this simple definition is not very robust 
        in cases when there are more than 2 attempts. Therefore, we will identify 
        them in a different way (as suggested by Diana), as follows:

        1. Sort batches by `production_date_formatted`. 
            This step is super important to ensure that earliest batches will 
            always be higher on the DataFrame!
        2. Identify batches that have duplicated `subject_id`, since those are 
            the ones that have more than one attempt.
        3. Group the batches by `subject_id` and add an ordinal number that
            represent the order in which they appeared. This will be the 
            batch's `attempt_number`.
        4. If a batch had an `attempt_number` of `1`, then it is straightforward
            and its `attempt_type` is `first`.
        5. For the rest of batches with multiple attempts, we check if the 
            batch had an apheresis using the column 
            `apheresis_spiked_welded_date_time`. If it has a non-NaN 
            apheresis date, we classify it as `reaph`, otherwise, we classify 
            it as `reman`.
        6. All other batches with a single attempt are given an 
            `attempt_number` of 1 and thus an `attempt_type` of `first`.

    Parameters
    ----------
    df: pandas DataFrame
        Original DataFrame.

    verbose : bool
        Define if verbose output will be printed (True) or not (False).
        Defaults to True.

    Returns
    -------
    df: pandas DataFrame
        Same as the input DataFrame but with added columns `attempt_number`
        and `attempt_type`.

    """
    if ('attempt_number'  not in df.columns) and ('attempt_type'  not in df.columns):

        # Check that necessary columns are present in the DataFrame.
        cols_needed = ['batch_id', 'subject_id', 'production_date_formatted']
        for col_ in cols_needed:
            if col_ not in df.columns:
                raise Exception(f"Column {col_} not present in df. Needed for calculating `attempt_number` and `attempt_number`.")

        # These columns are different for CLP and XLP. Thus, they 
        # need to be processed differently.
        cols_aph = ['apheresis_spiked_welded_date_time', # CLP
                    'apheresis_spike_date'] # XLP
        col_aph = None
        for col_aph_ in cols_aph:
            if col_aph_ in df.columns:
                col_aph = col_aph_
                break

        if verbose:
            print("+ Adding columns `attempt_number` and `attempt_number`...", end='', flush=True)

        # 1. Sort batches by `production_date_formatted`. 
        # This step is super important to ensure that earliest batches will 
        # always be higher on the DataFrame!
        df_sorted_date = df.sort_values(by='production_date_formatted')

        # 2. Identify batches that have duplicated `subject_id`, since those are 
        # the ones that have more than one attempt.
        df_batches_many_attempts = df_sorted_date[df_sorted_date.duplicated('subject_id', keep=False) == True]
        df_batches_many_attempts = df_batches_many_attempts.sort_values(by='production_date_formatted')

        # 3. Group the batches by `subject_id` and add an ordinal number that
        # represent the order in which they appeared. This will be the 
        # batch's `attempt_number`.
        df_batches_many_attempts['attempt_number'] = df_batches_many_attempts.groupby('subject_id').cumcount()+1


        def classify_attempt(row):

            # 4. If a batch had an `attempt_number` of `1`, then it is straightforward
            # and its `attempt_type` is `first`.    
            if row['attempt_number'] == 1:
                attempt = 'first'

            # 5. For the rest of batches with multiple attempts, we check if the 
            # batch had an apheresis using the column 
            # `col_aph`. If it has a non-NaN apheresis date, 
            # we classify it as `reaph`, otherwise, we classify it as `reman`.
            else:
                if row[col_aph] == row[col_aph]:
                    attempt = 'reaph'
                else:
                    attempt = 'reman'

            return attempt

        df_batches_many_attempts['attempt_type'] = df_batches_many_attempts.apply(classify_attempt, axis=1)


        # 6. All other batches with a single attempt are given an `attempt_number`
        # of 1 and thus an `attempt_type` of `first`.
        df = pd.merge(df, df_batches_many_attempts[['batch_id', 'attempt_number', 'attempt_type']], how="left", on=['batch_id'])
        df['attempt_number'] = df['attempt_number'].fillna(1)
        df['attempt_type'] = df['attempt_type'].fillna('first')

        if verbose:
            print("\tDONE!")

    else:
        if verbose:
            print(". Columns `attempt_number` and `attempt_type` already existed in the DataFrame.")

    return df
