    def calculate_kpis(df, verbose=True):
    """`calculate_kpis`

    Calculate basic set of KPIs.

    !!! info
        Additional KPIs are calculated in the function [`kipy.calculate_kpis2`][]

    Parameters
    ----------
    df : pandas DataFrame
        DataFrame with deviation information. In other words, 
        a DataFrame that passed already through the following functions:

        * [kipy.classify_deviation][]

        * [kipy.classify_deviation_extra][]

        * [kipy.count_deviations][]

    verbose : bool, optional
        Define if verbose output will be printed (True) or not (False).

    Returns
    -------
    df_kpis : pandas DataFrame
        A DataFrame with the following KPIs:

        | KPI Name       | Description                         |
        |----------------|-------------------------------------|
        | `n_batches`    | Number of batches                   |
        | `harvested_n`  | Number of harvested batches         |
        | `harvested_p`  | Percentage of harvested batches     |
        | `terminated_n` | Number of terminated batches        |
        | `terminated_p` | Percentage of terminated batches    |
        | `oos_n`        | Number of batches with OOS          |
        | `oos_p`        | Percentage of batches with OOS      |
        | `no_oos_n`     | Number of batches *without* OOS     |
        | `no_oos_p`     | Percentage of batches *without* OOS |

    See also
    --------
    * [`kipy.calculate_kpis2`][]
    """
    if verbose:
        print("Calculating set of KPIs...", end='', flush=True)

    kpis = {}

    kpis['n_batches'] = len(df)
    kpis['harvested_n'] = len(df.query('harvested == True'))
    kpis['harvested_p'] = round((kpis['harvested_n'] / kpis['n_batches']) * 100, 2)
    kpis['terminated_n'] = len(df.query('harvested == False'))
    kpis['terminated_p'] = round((kpis['terminated_n'] / kpis['n_batches']) * 100, 2)
    kpis['oos_n'] = len(df.query('oos == True'))
    kpis['oos_p'] = round((kpis['oos_n'] / kpis['n_batches']) * 100, 2)
    kpis['no_oos_n'] = kpis['n_batches'] - kpis['oos_n']
    kpis['no_oos_p'] = round((kpis['no_oos_n'] / kpis['n_batches']) * 100, 2)

    df_kpis = pd.DataFrame(kpis.items(), columns=['kpi', 'value'])

    if verbose:
        print("\tDONE!")

    return df_kpis

    def calculate_kpis2(df, date_end, verbose=True):
    """`calculate_kpis2`

    Calculate KPIs of interest in past and present.

    !!! info
        Basic KPIs are calculated in the function [`kipy.calculate_kpis`][]

    Parameters
    ----------
    df : pandas DataFrame
        DataFrame with deviation information. In other words, 
        a DataFrame that passed already through the following functions:

        * [kipy.classify_deviation][]

        * [kipy.classify_deviation_extra][]

        * [kipy.count_deviations][]

    verbose : bool, optional
        Define if verbose output will be printed (True) or not (False).

    Returns
    -------
    df_kpis : pandas DataFrame
        A DataFrame with the following KPIs:

        | KPI Name             | Description                                                        |
        |----------------------|--------------------------------------------------------------------|
        | `X_batches_n`        | Number of batches                                                  |
        | `X_fresh_n`          | Number of fresh batches                                            |
        | `X_fresh_p`          | Percentage of fresh batches                                        |
        | `X_frozen_n`         | Number of frozen batches                                           |
        | `X_frozen_p`         | Percentage of frozen batches                                       |
        | `X_terminated_n`     | Number of terminated batches (based on their status)               |
        | `X_terminated_p`     | Percentage of terminated batches (based on their status)           |
        | `X_oos_rejected_n`   | Number of batches with status "rejected" or "pending" with OOS     |
        | `X_oos_rejected_p`   | Percentage of batches with status "rejected" or "pending" with OOS |
        | `X_oos_physicians_n` | Number of batches with status "physician's release" with OOS       |
        | `X_oos_physicians_p` | Percentage of batches with status "physician's release" with OOS   |
        | `X_bags_n`           | Number of produced bags                                            |
        | `X_bags_vi_n`        | Number of bags with a VI value of `True`                           |
        | `X_bags_vi_p`        | Percentage of bags with a VI value of `True`                       |
        | `X_oos_batches_n`    | Number of batches with OOS                                         |
        | `X_oos_batches_p`    | Percentage of batches with OOS                                     |
        | `X_success_n`        | Number of batches that were successful                             |
        | `X_success_p`        | Percentage of batches that were successful                         |
        | `X_success_fresh_n`  | Number of *fresh* batches that were successful                     |
        | `X_success_fresh_p`  | Percentage of *fresh* batches that were successful                 |
        | `X_success_frozen_n` | Number of *frozen* batches that were successful                    |
        | `X_success_frozen_p` | Percentage of *frozen* batches that were successful                |

        where X can either be `past` or `present`.

    See also
    --------
    * [`kipy.calculate_kpis`][]
    """

    if verbose:
        print("Calculating set of past/present KPIs (KPIs 2)...")


    date_start, date_partition = misc.get_date_start_partition(date_end)
    if verbose:
        print(f"\tBeginning date: {date_start}")
        print(f"\tPartition date: {date_partition}")
        print(f"\tEnd date: {date_end}")

    # Divide data into past (i.e., historical) and present.
    if '-q' in str(date_end):
        df_past = df.query('(production_date_q >= @date_start) and (production_date_q <= @date_partition)').copy()
        df_present = df.query('production_date_q > @date_partition').copy()
    else:
        df_past = df.query('(production_date_short_formatted >= @date_start) & (production_date_short_formatted <= @date_partition)').copy()
        df_present = df.query('production_date_short_formatted > @date_partition').copy()


    kpis2 = {}

    # Loop, since calculations are done exactly the same for past and
    # present data.
    for df_, prefix in zip([df_past, df_present], ['past', 'present']):

        # Calculate the number of periods.
        # For past data, this is used to calculate the average.
        # For present data, this will always be one (we are looking at only
        # one period). However, we do it calculate it to keep it consistent
        # between both cases.
        if '-q' in str(date_end):
            periods = list(df_['production_date_q'].unique())
            n_periods = len(periods)
        else:
            periods = df_['production_date_short'].unique()
            n_periods = len(periods)

        # Initialize lists. 
        # Each element will correspond to the value of one period.
        n_batches = []
        n_batches_fresh = []
        p_batches_fresh = []
        n_batches_frozen = []
        p_batches_frozen = []
        n_batches_terminated = []
        p_batches_terminated = []
        n_oos_rejected = []
        p_oos_rejected = []
        n_oos_physicians = []
        p_oos_physicians = []
        n_bags = []
        n_bags_vi = []
        p_bags_vi = []
        n_oos = []
        p_oos = []
        n_success = []
        p_success = []
        n_success_fresh = []
        p_success_fresh = []
        n_success_frozen = []
        p_success_frozen = []

        for period in periods:

            if '-q' in str(date_end):
                df_period = df_.query('production_date_q == @period').copy()
            else:
                df_period = df_.query('production_date_short == @period').copy()

            # Number of total batches.
            n_batches_period = len(df_period)
            n_batches.append(n_batches_period)

            # Number of fresh batches.
            n_batches_fresh_period = len(df_period.query('apheresis_type == "fresh"'))
            n_batches_fresh.append(n_batches_fresh_period)
            p_batches_fresh_period = round((n_batches_fresh_period/n_batches_period)*100, 2)
            p_batches_fresh.append(p_batches_fresh_period)

            # Number of frozen batches.
            n_batches_frozen_period = len(df_period.query('apheresis_type == "frozen"'))
            n_batches_frozen.append(n_batches_frozen_period)
            p_batches_frozen_period = round((n_batches_frozen_period/n_batches_period)*100, 2)
            p_batches_frozen.append(p_batches_frozen_period)

            # Number of terminated batches.
            n_batches_terminated_period = len(df_period.query('status == "terminated"'))
            n_batches_terminated.append(n_batches_terminated_period)
            p_batches_terminated_period = round((n_batches_terminated_period/n_batches_period)*100, 2)
            p_batches_terminated.append(p_batches_terminated_period)

            # Number of rejected batches (with status either "rejected" or "pending") with OOS.
            n_oos_rejected_period = len(df_period.query('(oos == True) & ((status == "rejected") or (status == "pending"))'))    
            n_oos_rejected.append(n_oos_rejected_period)
            p_oos_rejected_period = round((n_oos_rejected_period/n_batches_period)*100, 2)
            p_oos_rejected.append(p_oos_rejected_period)

            # Number of batches with status "physician's release" with OOS.
            n_oos_physicians_period = len(df_period.query('(oos == True) & (status == "physician\'s release")'))    
            n_oos_physicians.append(n_oos_physicians_period)
            p_oos_physicians_period = round((n_oos_physicians_period/n_batches_period)*100, 2)
            p_oos_physicians.append(p_oos_physicians_period)

            # Number of bags
            df_bags = df_period[['vi_bag1_deviation', 'vi_bag2_deviation']]
            n_bags_period = (len(df_bags) * 2) - df_bags.isnull().sum().sum()
            n_bags.append(n_bags_period)

            # Number of bags with VI.
            n_bags_vi_period = df_bags.sum().sum()
            n_bags_vi.append(n_bags_vi_period)
            p_bags_vi_period = round((n_bags_vi_period/n_bags_period)*100, 2)
            p_bags_vi.append(p_bags_vi_period)

            # Number of batches with OOS.
            n_oos_period = len(df_period.query('oos == True'))
            n_oos.append(n_oos_period)
            p_oos_period = round((n_oos_period/n_batches_period)*100, 2)
            p_oos.append(p_oos_period)

            # Success rate.
            n_success_period = len(df_period.query('success == True'))
            n_success.append(n_success_period)
            p_success_period = round((n_success_period/n_batches_period)*100, 2)
            p_success.append(p_success_period)            

            # Success rate of fresh batches.
            n_success_fresh_period = len(df_period.query('(success == True) & (apheresis_type == "fresh")'))
            n_success_fresh.append(n_success_fresh_period)            
            p_success_fresh_period = round((n_success_fresh_period/n_batches_period)*100, 2)
            p_success_fresh.append(p_success_fresh_period)  

            # Success rate of frozen batches.
            n_success_frozen_period = len(df_period.query('(success == True) & (apheresis_type == "frozen")'))
            n_success_frozen.append(n_success_frozen_period)
            p_success_frozen_period = round((n_success_frozen_period/n_batches_period)*100, 2)
            p_success_frozen.append(p_success_frozen_period) 

        if n_periods > 0:
            kpis2[prefix + '_batches_n'] = round(sum(n_batches)/n_periods, 2)
            kpis2[prefix + '_fresh_n'] = round(sum(n_batches_fresh)/n_periods, 2)
            kpis2[prefix + '_fresh_p'] = round(sum(p_batches_fresh)/n_periods, 2)
            kpis2[prefix + '_frozen_n'] = round(sum(n_batches_frozen)/n_periods, 2)
            kpis2[prefix + '_frozen_p'] = round(sum(p_batches_frozen)/n_periods, 2)
            kpis2[prefix + '_terminated_n'] = round(sum(n_batches_terminated)/n_periods, 2)
            kpis2[prefix + '_terminated_p'] = round(sum(p_batches_terminated)/n_periods, 2)
            kpis2[prefix + '_oos_rejected_n'] = round(sum(n_oos_rejected)/n_periods, 2)
            kpis2[prefix + '_oos_rejected_p'] = round(sum(p_oos_rejected)/n_periods, 2)
            kpis2[prefix + '_oos_physicians_n'] = round(sum(n_oos_physicians)/n_periods, 2)
            kpis2[prefix + '_oos_physicians_p'] = round(sum(p_oos_physicians)/n_periods, 2)
            kpis2[prefix + '_bags_n'] = round(sum(n_bags)/n_periods, 2)
            kpis2[prefix + '_bags_vi_n'] = round(sum(n_bags_vi)/n_periods, 2)
            kpis2[prefix + '_bags_vi_p'] = round(sum(p_bags_vi)/n_periods, 2)
            kpis2[prefix + '_oos_batches_n'] = round(sum(n_oos)/n_periods, 2)
            kpis2[prefix + '_oos_batches_p'] = round(sum(p_oos)/n_periods, 2)
            kpis2[prefix + '_success_n'] = round(sum(n_success)/n_periods, 2)
            kpis2[prefix + '_success_p'] = round(sum(p_success)/n_periods, 2)
            kpis2[prefix + '_success_fresh_n'] = round(sum(n_success_fresh)/n_periods, 2)
            kpis2[prefix + '_success_fresh_p'] = round(sum(p_success_fresh)/n_periods, 2)
            kpis2[prefix + '_success_frozen_n'] = round(sum(n_success_frozen)/n_periods, 2)
            kpis2[prefix + '_success_frozen_p'] = round(sum(p_success_frozen)/n_periods, 2)
        else:
            kpis2[prefix + '_batches_n'] = np.nan
            kpis2[prefix + '_fresh_n'] = np.nan
            kpis2[prefix + '_fresh_p'] = np.nan
            kpis2[prefix + '_frozen_n'] = np.nan
            kpis2[prefix + '_frozen_p'] = np.nan
            kpis2[prefix + '_terminated_n'] = np.nan
            kpis2[prefix + '_terminated_p'] = np.nan
            kpis2[prefix + '_oos_rejected_n'] = np.nan
            kpis2[prefix + '_oos_rejected_p'] = np.nan
            kpis2[prefix + '_oos_physicians_n'] = np.nan
            kpis2[prefix + '_oos_physicians_p'] = np.nan
            kpis2[prefix + '_bags_n'] = np.nan
            kpis2[prefix + '_bags_vi_n'] = np.nan
            kpis2[prefix + '_bags_vi_p'] = np.nan
            kpis2[prefix + '_oos_batches_n'] = np.nan
            kpis2[prefix + '_oos_batches_p'] = np.nan
            kpis2[prefix + '_success_n'] = np.nan
            kpis2[prefix + '_success_p'] = np.nan
            kpis2[prefix + '_success_fresh_n'] = np.nan
            kpis2[prefix + '_success_fresh_p'] = np.nan
            kpis2[prefix + '_success_frozen_n'] = np.nan
            kpis2[prefix + '_success_frozen_p'] = np.nan

    df_kpis2 = pd.DataFrame(kpis2.items(), columns=['kpi2', 'value'])

    if verbose:
        print("\t\t\tDONE!")

    return df_kpis2
