    def preprocess_raw_dataframe(df, product, path_dictionary, verbose=True):
    """`preprocess_raw_dataframe`

    Pre-process a raw `CLP`/`XLP` DataFrame by applying the next steps:

    * Rename columns (according to the given data dictionary mapping)
    * Cleaning of numerical and categorical columns

    Parameters
    ----------
    df : pandas DataFrame
        Raw DataFrame (as read using [`kipy.read_excel_file_raw`][])

    product : str
        The product of interest. It accepts the following values:

        * `yescarta` or `clp`
        * `tecartus` or `xlp`

    path_dictionary : str, pathlib.Path
        Location of the `.xlsx` data dictionary.

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).


    Returns
    -------
    df_preprocessed : pandas DataFrame
        Pre-processed DataFrame.

    See also
    ----------
    * [`kipy.rename_columns`][]
    * [`kipy.clean_dataframe`][]
    """    

    # Since verbose is handled internally for each function, there's 
    # no need to add any extra prints here.
    df_renamed = rename_columns(df, product, path_dictionary, verbose=verbose)   
    df_preprocessed = clean_dataframe(df_renamed, product, path_dictionary, verbose=verbose)

    return df_preprocessed

    def rename_columns(df, product, path_dictionary, verbose=True):
    """`rename_columns`

    Rename the columns so that they have computer names as defined
    in the corresponding data dictionary.

    !!! tip
        Renaming the columns should always be the first pre-processing step.

    Parameters
    ----------
    df : pandas DataFrame
        The pandas DataFrame.

    product : str
        The product of interest. It accepts the following values:

        * `yescarta` or `clp`
        * `tecartus` or `xlp`

    path_dictionary : str, pathlib.Path
        Location of the `.xlsx` data dictionary.

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    df_renamed : pandas DataFrame
        Same as input `df`, but with the columns renamed according
        to the data dictionary mapping.
    """
    if verbose:
        print("Renaming columns...", flush=True, end='')

    # Make sure that the given path is a pathlib.Path.
    path_dictionary = pathlib.Path(path_dictionary)

    # Check that data dictionary exists.
    if not path_dictionary.exists():
        raise Exception(f"Data dictionary {path_dictionary} does not exist.")

    # Read the dictionary.
    df_dictionary = pd.read_excel(path_dictionary, 
                                  sheet_name='dictionary', 
                                  header=0)

    # Select the columns of interest.
    df_dictionary = df_dictionary[['Column Name', 'Computer Column Name']]

    # Convert DataFrame to dictionary.  
    df_dictionary_dict = dict(zip(df_dictionary['Column Name'].values, 
                                  df_dictionary['Computer Column Name'].values))

    # Perform the renaming.
    df_renamed = df.rename(columns=df_dictionary_dict)

    if verbose:
        print("\tDONE!")

    return df_renamed

    def clean_dataframe(df, product, path_dictionary, cols=None, verbose=True):
    """`clean_dataframe`

    Clean categorical and numerical columns of a `CLP`/`XLP` DataFrame.

    !!! info
        To identify them, it uses information from the given data dictionary.
        Numerical columns are those that have a `Type` of `int`, `float`, or
        `int (scientific notation)`. Categorical columns are those that have a 
        `Type` of `string`. Columns that fall outside of these types remain
        unchanged.


    Parameters
    ----------
    df : pandas DataFrame
        The pandas DataFrame.

    product : str
        The product of interest. It accepts the following values:

        * `yescarta` or `clp`
        * `tecartus` or `xlp`

    path_dictionary : str, pathlib.Path
        Location of the `.xlsx` data dictionary.

    cols : list of str
        List with the columns to be cleaned. If `None`, 
        all columns will be (attempted to be) cleaned.

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    df_clean : pandas DataFrame
        Clean DataFrame.

    See also
    ----------
    * [`kipy.clean_column_numerical`][]
    * [`kipy.clean_column_categorical`][]
    """

    if verbose:
        print("Cleaning columns...")

    # Make sure that the given path is a pathlib.Path.
    path_dictionary = pathlib.Path(path_dictionary)

    # Check that data dictionary exists.
    if not path_dictionary.exists():
        raise Exception(f"Data dictionary {path_dictionary} does not exist.")

    # Read the dictionary.
    df_dictionary = pd.read_excel(path_dictionary, 
                                  sheet_name='dictionary', 
                                  header=0)

    # Select the columns of interest.
    df_dictionary = df_dictionary[['Computer Column Name', 'Unit', 'Type']]
    df_dictionary = df_dictionary.set_index('Computer Column Name')


    # Define which columns will be cleaned.
    if cols is None:
        cols = df.columns

    # Perform cleaning of columns.
    # This is done one by one and depending on the column type.
    types_numerical = ['int', 'float', 'int (scientific notation)']
    types_categorical = ['string']

    df_clean = df.copy()
    for col in cols:
        col_unit = str(df_dictionary.loc[col, 'Unit']).lower()
        col_type = str(df_dictionary.loc[col, 'Type']).lower()

        if col_type in types_numerical:
            if verbose:
                print(f"+ Cleaning numerical column {col}...", end='', flush=True)
            df_clean[col] = clean_column_numerical(df, col)

            if col_unit == '%':
                df_clean[col] = df_clean[col] * 100

            if verbose:
                print("\t DONE!")

        elif col_type in types_categorical:
            if verbose:
                print(f"+ Cleaning categorical column {col}...", end='', flush=True)
            df_clean[col] = clean_column_categorical(df, col)
            if verbose:
                print("\t DONE!")
        else:
            if verbose:
                print(f"- Column {col} will not be cleaned and left as is.")


    if verbose:
        print("\tDONE!")


    return df_clean

    clean_column_numerical

    Clean a numerical column. It applies the following steps:

    • Convert empty spaces (i.e., ' ') to NaNs.
    • Convert entries with a value of 'No Result' to NaNs.
    • Convert weird entries with a value of \xa0 to NaNs.
    • Convert entries with a value of 'nan' to NaNs.
    • Convert LOQs (i.e., entries that have a < such as <LOQ or <8) to -1.
    • Convert entries with a value of #VALUE! or #DIV/0! to NaNs.

    • Cast to float to ensure that values will be numbers.

    Danger

    If the LOQ definition changes in the future, this function needs to be updated.

    def clean_column_numerical(df, col):
    """`clean_column_numerical`

    Clean a numerical column. It applies the following steps:

    * Convert empty spaces (i.e., `' '`) to `NaN`s.
    * Convert entries with a value of `'No Result'` to `NaN`s.
    * Convert weird entries with a value of `\\xa0` to `NaN`s.
    * Convert entries with a value of `'nan'` to `NaN`s.
    * Convert `LOQ`s (i.e., entries that have a `<` such as `<LOQ` or `<8`) 
      to `-1`.
    * Convert entries with a value of `#VALUE!` or `#DIV/0!` to `NaN`s.
    * Remove the percentage character.
    * Cast to float to ensure that values will be numbers.

    !!! warning

        Notice that percentage entries are cleaned by removing the `%` only.
        For example, `75%` is cleaned as `0.75`. If an output of `75` is
        needed, it needs to be multiplied by `100` outside of this function
        (for example, as done in [`kipy.clean_dataframe`][])

    !!! danger

        If the `LOQ` definition changes in the future, this function needs 
        to be updated.

    Parameters
    ----------
    df : pandas DataFrame
        The pandas DataFrame.s

    col : string
        Name of the numerical column to be cleaned.

    Returns
    -------
    col_clean: pandas Series
        The clean (numerical) column.

    See also
    ----------
    * [`kipy.clean_dataframe`][]
    """
    df_clean = df.copy(deep=True)

    # Clean weird string entries.
    df_clean.loc[df[col]==' ', col] = np.nan
    df_clean.loc[df[col]=='No Result', col] = np.nan
    df_clean.loc[df[col]=='\xa0', col] = np.nan
    df_clean.loc[df[col]=='nan', col] = np.nan

    # Deal with LOQs
    df_clean.loc[df[col]=='<LOQ', col] = -1
    df_clean.loc[df[col]=='<10%', col] = -0.01 # Since this is a % column, it will be multiplied by 100.
    df_clean.loc[df_clean[col].astype(str).str.contains('<'), col] = -1

    # In case we want to clean LOQs diffrently (e.g., <10 --> 10)
    # df_clean[col] = df_clean[col].astype(str).str.replace('<', '')


    # Deal with wrong (Excel) formula results.
    df_clean.loc[df[col]=='#VALUE!', col] = np.nan
    df_clean.loc[df[col]=='#DIV/0!', col] = np.nan


    # Remove percentage character.
    df_clean[col] = df_clean[col].astype(str).str.replace('%', '')

    # Cast to float (i.e., ensure that it will be a number).
    df_clean[col] = df_clean[col].astype(float)

    return df_clean[col]

    def clean_column_categorical(df, col):
    """`clean_column_categorical`

    Clean a categorical column. It applies the following steps:

    * Make strings lower case
    * Remove leading spaces
    * Remove trailing spaces
    * Convert weird entries with a value of `\\xa0` to `NaN`s.

    Parameters
    ----------
    df : pandas DataFrame
        The pandas DataFrame.

    col : string
        Name of the numerical column to be cleaned.

    Returns
    -------
    col_clean: pandas Series
        The clean (numerical) column.

    See also
    ----------
    * [`kipy.clean_dataframe`][]
    """

    df_clean = df.copy(deep=True)

    # Make lower case and remove leading/trailing spaces.
    def _clean_string(string):

        if isinstance(string, str):
            clean_string = string.lower().strip()
        else:
            clean_string = string

        return clean_string
    df_clean[col] = df_clean[col].apply(_clean_string)

    # Remove weird string entries.
    df_clean.loc[df[col]=='\xa0', col] = np.nan

    return df_clean[col]
