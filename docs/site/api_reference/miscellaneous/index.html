    def get_date_end(time_resolution):
    """`get_date_end`

    Based on the current day (i.e., the day when the script is being run), 
    get the end date of the data to be used.

    !!! example

        | Current date | `time_resolution` | `date_end`            | `date_end` format |
        |--------------|-------------------|-----------------------|-------------------|
        | May 15, 2023 | `monthly`         | `2023-04-30 23:59:59` | [pandas timestamp](https://pandas.pydata.org/pandas-docs/stable/reference/api/pandas.to_datetime.html)  |
        | May 15, 2023 | `quarter`         | `'2023-q1'`           | string            |

    Parameters
    ----------
    time_resolution : str
        Time resolution of the desired `latest_date`.
        There are two possible options:

        * `'month'`: the end date is given as the last day of the month 
        previous to the day when the script is being run.

        * `'quarter'`: the end date is the quarter previous to the
        quarter of the day when the script is being run.

    Returns
    -------
    date_end : pandas datetime or str
        Date delimiting the data to be used.

    See also
    --------
    * [`kipy.get_date_start_partition`][]
    """

    if time_resolution == 'month':
        end_date = pd.to_datetime('today') - pd.offsets.MonthEnd(1)

        # Make sure the latest date is at 23:59:59    
        end_date = end_date.replace(hour=23, minute=59, second=59, microsecond=0)


    elif time_resolution == 'quarter':
        end_date = pd.to_datetime('today').to_period('Q') - 1
        end_date = str(end_date.year) + '-q' + str(end_date.quarter)

    else:
        raise Exception(f"{time_resolution} is not a valid value for time_resolution.")


    return end_date

    def get_date_start_partition(end_date, offset=None):
    """`get_date_start_partition`

    Based on the given `end_date`, get a `date_start` (i.e., the starting date
    of the data to be used) and a `date_partition` (i.e., the date that 
    partitions the date between historical data and present data).

    Parameters
    ----------
    end_date : str
        End date of the data to be used. It has the following format:

        * For a `time_resolution` of `month`, `2023-10-31 23:59`

        * For a `time_resolution` of `quarter`, `2023-q3`

        It can be obtained using the function [`kipy.get_date_end`][].

    offset : int
        Offset to consider for defining `date_start` before `date_partition`.

        * If `end_date` is defined monthly, `offset` defaults to 12, 
        which corresponds to a year.

        * If `end_date` is defined quarterly, `offset` defaults to 3,
        which corresponds to a year (due to how quarter are defined, since
        they don't encompass individual days).


    Returns
    -------
    date_start, date_partition : same type as date_end
        Dates that mark the start and the partition of the data.

        !!! example

            | Current date | Current quarter | End date              | `date_start`          | `date_partition`      |
            |--------------|-----------------|-----------------------|-----------------------|-----------------------|
            | May 15, 2023 | 2023-Q2         | `2023-04-30 23:59:59` | `2022-04-01 00:00:00` | `2023-03-31 23:59:59` |
            |              |                 | `2023-q2`             | `2022-q2`             | `2023-q1            ` |


    See also
    --------
    * [`kipy.get_date_end`][]

    """
    # The partition date allows dividing batches from the current (previous)
    # period of interest and the past.
    # For example, if we are are on 2023-06-XX, the latest_date will be
    # 2023-05-31. Then, partition_date will be 
    # * 2023-04-30 for monthly
    # * 2023-q1 for quarter
    if '-q' in str(end_date):
        partition_date = pd.to_datetime(end_date).to_period('Q') - 1
        partition_date = str(partition_date.year) + '-q' + str(partition_date.quarter)

        if offset is None:
            offset = 3
    else:
        partition_date = end_date - pd.offsets.MonthEnd(1)

        if offset is None:
            offset = 12

    # The beginning_date is 4 quarters/12 months before the partition date, since
    # that is the time period that we are using as a reference now.
    # We subtract115 months and go back to the begining of that month,
    # making it 12 effectively. For example, if partition date is 2023-04-30, 
    # then beginnning_date will be 
    # * 2022-05-01 for monthly
    # * 2022-q1 for quarter
    if '-q' in str(end_date):
        beginning_date = pd.to_datetime(partition_date).to_period('Q') - offset
        beginning_date = str(beginning_date.year) + '-q' + str(beginning_date.quarter)
    else:
        beginning_date = end_date - pd.offsets.MonthEnd(offset)
        beginning_date = beginning_date.replace(day=1, hour=00, minute=00, second=00, microsecond=0)

    return beginning_date, partition_date

    def save_df_as_csv(df, file_path, sep=';', decimal=',', verbose=True):
    """`save_df_as_csv`

    Wrapper to make sure that all `.csv` result files are saved in the same 
    way. 

    !!! tip
        In case we decide to change how to save `.csv` files, we can do so 
        in a single place.

    Parameters
    ----------
    df : pandas DataFrame
        DataFrame to be saved.

    file_path : string, pathlib.Path
        Path where the file will be saved. It includes the file name.

    sep : string
        Character used as a column separator for the `.csv` file.

        !!! tip
            For Power BI dashboards, using `';'` is recommended!
            Make sure to configure the settings properly to align with this 
            too.

    decimal : string
        Character used as decimal separator for the `.csv` file.

        !!! tip
            For Power BI dashboards, using `','` is recommended!
            Make sure to configure the settings properly to align with this 
            too.

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    None
    """

    # Ensure path is a pathlib.Path (and not just a string).
    file_path = pathlib.Path(file_path)

    # Actually save the file.
    df.to_csv(file_path, index=False, sep=sep, decimal=decimal)

    if verbose:
        print(f"{file_path} was saved successfully (separator = {sep}, decimal = {decimal}).")

    return None

    def get_elements_with_substring(base_list, substr_list):
    """ `get_elements_with_substring`

    Get elements of a list that have a specific substring.

    !!! tip
        This is a handy function for process monitoring. It is used 
        for extracting column names of a specific type (e.g., columns that 
        are OOS and their names in the DataFrame end with `_oos`).

    Parameters
    ----------
    base_list : list
        Base lists (for example, `list(df.columns)`)

    substr_list : list
        Each element of the list is a substring to be found. 

        !!! tip
            If interested in only one substring, pass a list with one element.

    Returns
    -------
    list
        List with the elements that have the given substring.
        If none, returns an empty list.

    References
    ----------
    * [Filtering a list of strings based on a substring](https://www.geeksforgeeks.org/python-filter-list-of-strings-based-on-the-substring-list/)
    """

    return [str for str in base_list if
             any(sub in str for sub in substr_list)]

    def group_column_in_time(df, col_name, col_period, col_year='production_date_year', date_start=None, date_end=None, verbose=True):
    """`group_column_in_time`

    Group the column of a DataFrame by counting the number and percentage
    of batches per period (either month or quarter). It also allows to be
    filtered by `date_start` and `date_end`.

    !!! example

        If `col_name` is `'harvested'` (which has two possible values:
        `True` and `False`), `col_period` is `'production_date_month'`,
        and `col_year` is `'production_date_year'`, `df_final` look as follows:

        | `harvested` | `production_date_month` | `production_date_year` | `n`   | `percentage` | `production_date_month_name` | `production_date_short` | `production_date_short_formatted ` |
        |-------------|-------------------------|------------------------|-------|--------------|------------------------------|-------------------------|------------------------------------|
        | `True`      | `01`                    | `2022`                 | `10`  | `50.00`      | `'January'`                  | `'2023-01'`             | `2023-01-01 00:00:00`              |
        | `True`      | `02`                    | `2022`                 | `15`  | `75.00`      | `'February'`                 | `'2023-02'`             | `2023-02-01 00:00:00`              |
        | `...`       | `...`                   | `...`                  | `...` | `...`        | `...`                        | `...`                   | `...`                              |
        | `True`      | `12`                    | `2022`                 | `20`  | `100.00`     | `'December'`                 | `'2023-12'`             | `2023-12-01 00:00:00`              |
        | `False`     | `01`                    | `2022`                 | `10`  | `50.00`      | `'January'`                  | `'2023-01'`             | `2023-01-01 00:00:00`              |
        | `False`     | `02`                    | `2022`                 | `5`   | `25.00`      | `'February'`                 | `'2023-02'`             | `2023-02-01 00:00:00`              |
        | `...`       | `...`                   | `...`                  | `...` | `...`        | `...`                        | `...`                   | `...`                              |
        | `False`     | `12`                    | `2022`                 | `0`   | `0.00`       | `'December'`                 | `'2023-12'`             | `2023-12-01 00:00:00`              |


    Parameters
    ----------
    df : pandas DataFrame
        Original DataFrame.

    col_name : str
        Name of the column to be grouped.

    col_period : str
        Name of the column in the DataFrame that has the period information.
        Possible values are:

            * `'production_date_month '`

            * `'production_date_quarter'`

    col_year : str
        Name of the column in the DataFrame that has the production year data.

        !!! info
            It defaults to `'production_date_year'` as given by the
            function [`kipy.rename_columns`][] and the default data dictionary.

    date_start : pandas datetime (for 'month') or str (for 'quarter')
        If different from `None`, date to be used for filtering.

    date_end : pandas datetime (for 'month') or str (for 'quarter')
        If different from `None`, date to be used for filtering.

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    df_polished : pandas DataFrame
        DataFrame with the following columns:

        | Column name                       | Description                                                       |
        |-----------------------------------|-------------------------------------------------------------------|
        | `col_name`                        | Same name as `'col_name'`                                         |
        | `col_period`                      | Same name as `'col_period'`                                       |
        |                                   | * If `production_date_month`, it contains months (01, 02, etc.)   |
        |                                   | * If `production_date_quarter`, it contains quarte (q1, q2, etc.) |
        | `col_year`                        | Same name as `'col_year'`                                         |
        | `n`                               | Number of batches for that combination                            |
        | `percentage`                      | Percentage of batches for that combination (with two decimals)    |
        | `production_date_month_name`      | Name of the production month                                      |
        | `production_date_short`           | Production date (short)                                           |
        | `production_date_short_formatted` | Produciton date short (in proper format)                          |

        `n` and `percentage` correspond to the batches in that time period.

    References
    ----------
    * [pandas value_counts](https://pandas.pydata.org/pandas-docs/dev/reference/api/pandas.DataFrame.value_counts.html#pandas.DataFrame.value_counts) (from v. 1.1)
    """

    if verbose:
        print(f"Grouping DataFrame by column `{col_name}` and period `{col_period}`...", end="", flush=True)

    # First, we will group and count.
    # Notice how we reset the index to make sure that we have a proper DataFrame.
    series_grouped = df.value_counts(subset=[col_name, col_period, col_year]) 
    df_grouped = pd.DataFrame({'n': series_grouped}).reset_index()

    # Additionally, we will calculate the percentage of events (and not just
    # the absolute number) to make our life easier in Power BI.
    df_grouped['n_percentage'] = df_grouped['n']/df_grouped.groupby([col_period, col_year])['n'].transform('sum')
    df_grouped['n_percentage'] = round(df_grouped['n_percentage'] * 100, 2)


    # Padding
    # Finally, we will create a padded data frame to ensure that all 
    # possible combinations (column values, periods, years) have a value 
    # (even if 0).
    col_values_unique = sorted(df_grouped[col_name].unique())
    if '_q' in col_period:
        periods_unique = ['1', '2', '3', '4']
    elif '_m' in col_period:
        periods_unique = ['01', '02', '03', '04', '05', '06', '07', '08', '09', '10', '11', '12']
    years_unique = sorted(df_grouped[col_year].unique())

    possible_combinations = list(itertools.product(col_values_unique, periods_unique, years_unique))
    df_padded = pd.DataFrame(possible_combinations, columns=[col_name, col_period, col_year]) # Rename columns

    # Fill in the padded data frame (missing values will be replaced by 0).
    df_filled = df_padded.merge(df_grouped, how='outer', on=[col_name, col_period, col_year])
    df_filled.fillna(value=0, inplace=True)


    # Polish DataFrame for use in Power BI
    df_polished = df_filled.copy()

    # For quarter, add column `production_date_q` (e.g., 2023-q2)
    if '_q' in col_period:
        df_polished['production_date_q'] = df_polished[col_year] + '-q' + df_polished[col_period]

    # For month, add column `production_date_month_name` (e.g., May)
    elif '_m' in col_period:

        def _month_number_to_name(row):
            if pd.isna(row['production_date_month']):
                month_name = '-'
            else:
                month_name = calendar.month_name[int(row['production_date_month'])]
            return month_name

        df_polished['production_date_month_name'] = df_polished.apply(_month_number_to_name, axis=1)



    # Create production_date_short column only in the month case.
    if '_m' in col_period:
        df_polished['production_date_short'] = df_polished[col_year] + '-' + df_polished[col_period]
        df_polished['production_date_short_formatted'] = pd.to_datetime(df_polished['production_date_short'], format='%Y-%m')

    # Filter data to make sure that it contains batches only in the 
    # time period of interest.
    if '_q' in col_period:
        if date_start:
            df_polished = df_polished.query('production_date_q >= @date_start').copy()
        if date_end:
            df_polished = df_polished.query('production_date_q <= @date_end').copy()

    elif '_m' in col_period:
        if date_start:
            df_polished = df_polished.query('production_date_short_formatted >= @date_start').copy()
        if date_end:
            df_polished = df_polished.query('production_date_short_formatted <= @date_end').copy()

    if verbose:
        print("\tDONE!")

    return df_polished
