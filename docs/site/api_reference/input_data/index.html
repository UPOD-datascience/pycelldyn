    def copy_most_recent_data_file(product, file_type, path_origin, path_destiny, verbose=True):
    """ `copy_most_recent_data_file`

    Create a time-stamped copy of the most recent data file
    in the required location.

    Parameters
    ----------
    product : str
        The product of interest. It accepts the following values:

        * `yescarta` or `clp`
        * `tecartus` or `xlp`

    file_type : str
        File type. It accepts the following values:

        * `working`: Working data file
        * `master`: Master data file

    path_origin : str, pathlib.Path
        Location of the original (usually the most recent) `CLP`/`XLP` file.

    path_destiny : str, pathlib.Path
        Location where the new copy of the `CLP`/`XLP` file will be saved.
        The name of the copied file will include the current date as
        a prefix. For instance, the file `Working CLP Data.xlsx` will
        be saved as `YYYY-MM-DD CLP Working Data.xlsx`.

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    new_file_path : pathlib.Path
        Path of the copied data file. Includes the file name and its 
        extension. If no original data file was found, `new_file_path`
        is `None`.

    """

    # Ensure that paths are pathlib.Paths.
    path_origin = pathlib.Path(path_origin)
    path_destiny = pathlib.Path(path_destiny)

    # Convert inputs for ease of use.
    if (product == 'yescarta') or (product=='clp'):
        product = 'clp'
    elif product == 'tecartus' or (product=='xlp'):
        product = 'xlp'
    else:
        raise Exception(f"Product {product} is not valid.")


    # Cycle through all the data files in path origin until a match is found.
    # If no match is found, file_interest will be None.
    # Notice that we will only catch the first file that is found.
    # In the current file structure, this works fine, but if that changes,
    # this function might need to be adapted.
    file_interest = None

    for file in path_origin.rglob('*.xlsx'):

        # Convert to string in lower case.
        file_name = str(file.stem).lower()

        if (product in file_name) & (file_type in file_name):
            file_interest = file
            if verbose:
                print(f"File {file_interest} found!")
            break

    if file_interest is None:
        if verbose:
            print(f"No file found in {path_origin} that matches product {product} and type {file_type}")
        new_file_path = None
    else:
        date_today = datetime.today().strftime('%Y-%m-%d')
        new_file_name = f'{date_today} {product.upper()} {file_type.capitalize()} Data.xlsx'

        # Ensure that path_destiny exists.
        if not path_destiny.exists():
            if verbose:
                print(f"\tDirectory {path_destiny} does not exist. Creating...", flush=False, end='')
            path_destiny.mkdir(parents=True)
            if verbose:
                print("\tDONE!")

        new_file_path = pathlib.Path(path_destiny, new_file_name)

        if verbose:
            print(f"Copying file from {file_interest} to {new_file_path}...", flush=True, end="")
        shutil.copyfile(file_interest, new_file_path)
        if verbose:
            print("\tDONE!")

    return new_file_path

    def read_excel_file_raw(path_to_file, verbose=True):
    """`read_excel_file_raw`

    Read a raw `CLP`/`XLP` Excel data file.

    !!! warning

        Notice that there's a different behaviour when reading from a `.csv` 
        compared to when reading from a `.xlsx` (i.e., Excel) file. For example, 
        when reading from a `.csv` file, columns that end with a percentage 
        will be read as strings (e.g., `'75%'`); when reading from an Excel file, 
        the same columns will be read as floats (e.g., `0.75`).

    Parameters
    ----------
    path_to_file : str, pathlib.Path
        Path to the desired `CLP` or `XLP` Excel data file. 
        Should include the extension (e.g., `filename.xlsx`).

    verbose : bool
        Define if verbose output will be printed (`True`) or not (`False`).

    Returns
    -------
    df: pandas DataFrame
        Raw DataFrame.  
    """

    # Make sure that path_to_file is a pathlib.Path
    path_to_file = pathlib.Path(path_to_file)
    file_name = path_to_file.name


    # CLP (YESCARTA)
    if ('clp' in str(file_name).lower()) or ('yescarta' in str(file_name).lower()):
        if verbose:
            print(f"Reading YESCARTA file {file_name}... ", end='', flush=True)

        df = pd.read_excel(path_to_file, sheet_name=0)

        if verbose:
            print("\tDONE!")


    # XLP (TECARTUS)
    # The XLP file has an additional row on top. Thus, the header = 1.
    elif ('xlp' in str(file_name).lower()) or ('tecartus' in str(file_name).lower()):
        if verbose:        
            print(f"Reading TECARTUS {file_name}... ", end='', flush=True)

        df = pd.read_excel(path_to_file, sheet_name=0, header=1)

        if verbose:
            print("\tDONE!")


    # Unknown file.
    else:
        print("File {file_name} doesn't specify if CLP or XLP. Will be read with header=0")

        if verbose:        
            print(f"Reading unknown {file_name}... ", end='', flush=True)

        df = pd.read_excel(path_to_file, sheet_name=0)

        if verbose:
            print("\tDONE!")

    return df
